  <script>

    const games = [

      {

        name: "Retro Bowl",

        game_url: "https://hfmanor.com/retro-bowl",

        game_image_icon: "https://hfmanor.com/image/retro-bowl",

        id: "retro-bowl"

      },

      {

        name: "Subway Surfers",

        game_url: "https://hfmanor.com/subway-surfers",

        game_image_icon: "https://hfmanor.com/image/subway-surfers",

        id: "subway-surfers"

      },

      {

        name: "Basketball Stars",

        game_url: "https://hfmanor.com/basketball-stars",

        game_image_icon: "https://hfmanor.com/image/basketball-stars",

        id: "basketball-stars"

      },

      {

        name: "Basket Random",

        game_url: "https://hfmanor.com/basket-random",

        game_image_icon: "https://hfmanor.com/image/basket-random",

        id: "basket-random"

      }

    ];


    const gameListEl = document.getElementById('game-list');

    const gamePageEl = document.getElementById('game-page');

    const iframeContainer = document.getElementById('game-iframe-container');

    const backBtn = document.getElementById('back-btn');

    const openBtn = document.getElementById('open-btn');


    let currentGame = null;

    // Populate

    games.forEach(game => {

      const card = document.createElement('div');

      card.className = 'game-card';

      card.innerHTML = `

        <img src="${game.game_image_icon}" alt="${game.name}" onerror="this.style.opacity=.5; this.nextElementSibling.style.opacity=.7;">

        <h2>${game.name}</h2>

      `;

      card.addEventListener('click', () => showGame(game));

      gameListEl.appendChild(card);

    });


    function showGame(game) {

      currentGame = game;

      iframeContainer.innerHTML = '';

      // create iframe element safely instead of innerHTML

      const iframe = document.createElement('iframe');

      iframe.src = game.game_url;

      iframe.setAttribute('allowfullscreen', '');

      iframe.title = game.name;

      iframeContainer.appendChild(iframe);

      // UI

      gameListEl.style.display = 'none';

      gamePageEl.classList.add('active');

      // set open button href behavior

      openBtn.onclick = () => window.open(game.game_url, '_blank', 'noopener');

    }


    backBtn.addEventListener('click', () => {

      gamePageEl.classList.remove('active');

      gameListEl.style.display = 'grid';

      iframeContainer.innerHTML = '';

      currentGame = null;

    });

  </script>
